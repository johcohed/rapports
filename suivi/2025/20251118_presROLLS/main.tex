\begin{frame}{Le projet}
\textbf{Mes buts :}

\begin{itemize}
  \item Reproduire des rouleaux dans un \textbf{cadre idéalisé} (travail de Wahiba et Thomas)
  \item Influence des paramètres (profil de température, flux de surface, hétéréogénéité des flux, vents) sur les rouleaux (taille, formation, durée de vie)
  \item Evaluer transports de moments par les rouleaux ? (\cite{gryschkaImpactForcedRoll2014}) 
  \item Paramétrisation des rouleaux ? (\cite{fosterWhyRollsAre2005},  \cite{wangDeepLearningParameterization2023})
  \item Avoir descriptions moodèle forward des rouleaux pour ensuite faire le problème inverse et récupérer des flux ? 
  \end{itemize}
  \textbf{Etat d'avancement:}
  \begin{itemize}
    \item  Test pour obtenir des rouleaux avec des domaines idéeaux et simples. Lecture de littérature. 
  \end{itemize}
\end{frame}
